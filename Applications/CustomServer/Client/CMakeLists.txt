PROJECT(csClient)

CMAKE_MINIMUM_REQUIRED(VERSION 2.2)

INCLUDE_DIRECTORIES(
  ${csClient_SOURCE_DIR}
  ${csClient_BINARY_DIR}
)

IF(VTK_USE_MPI)
  INCLUDE_DIRECTORIES(${MPI_INCLUDE_PATH})
ENDIF(VTK_USE_MPI)

QT4_WRAP_CPP(MOC_BUILT_SOURCES
  MainWindow.h
  ${PYTHON_MOC_SOURCES}
)

QT4_WRAP_UI(UI_BUILT_SOURCES
)

SOURCE_GROUP("Generated" FILES
  ${MOC_BUILT_SOURCES}
  ${UI_BUILT_SOURCES}
)

IF(WIN32)
  LINK_DIRECTORIES(${QT_LIBRARY_DIR})
ENDIF(WIN32)

ADD_EXECUTABLE(csClient WIN32 MACOSX_BUNDLE
  MainWindow.cxx
  MainWindow.h
  main.cxx
  ProcessModuleGUIHelper.cxx
  ProcessModuleGUIHelper.h
  ${MOC_BUILT_SOURCES}
  ${UI_BUILT_SOURCES}
)

TARGET_LINK_LIBRARIES(csClient
  pqWidgets
  QtChart
  QtTesting
  QtWidgets
  ${QT_XML_LIBRARIES}
)

IF(WIN32)
  TARGET_LINK_LIBRARIES(csClient
    optimized qtmain debug qtmaind
  )
ENDIF(WIN32)
