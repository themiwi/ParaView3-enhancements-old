PROJECT(Xdmf)

SET(HDF5_LIBRARY vtkhdf5)
IF(VTK_USE_SYSTEM_EXPAT)
  SET(XDMF_EXPAT_LIBRARIES EXPAT_LIBRARY)
ELSE(VTK_USE_SYSTEM_EXPAT)
  SET(XDMF_EXPAT_LIBRARIES vtkexpat)
ENDIF(VTK_USE_SYSTEM_EXPAT)
IF(VTK_USE_ANSI_STDLIB MATCHES "^VTK_USE_ANSI_STDLIB$")
  SET(XDMF_USE_ANSI_STDLIB 1)
ELSE(VTK_USE_ANSI_STDLIB MATCHES "^VTK_USE_ANSI_STDLIB$")
  SET(XDMF_USE_ANSI_STDLIB ${VTK_USE_ANSI_STDLIB})
ENDIF(VTK_USE_ANSI_STDLIB MATCHES "^VTK_USE_ANSI_STDLIB$")

INCLUDE_DIRECTORIES(
  ${Xdmf_SOURCE_DIR}/Ice
  ${Xdmf_BINARY_DIR}/Ice
  ${Xdmf_SOURCE_DIR}/libsrc
  ${Xdmf_BINARY_DIR}/libsrc
  ${Xdmf_SOURCE_DIR}/vtk
  ${Xdmf_BINARY_DIR}/vtk
  ${VTK_INCLUDE_DIR}
  ${HDF5_INCLUDE_DIR}
  )
SUBDIRS(Ice)
SUBDIRS(libsrc)
SUBDIRS(vtk)
