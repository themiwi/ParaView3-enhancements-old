PROJECT(pqPython)

FIND_PACKAGE(PythonLibs)

INCLUDE_DIRECTORIES(
 ${ParaView_SOURCE_DIR}/VTK/Common
 ${ParaView_BINARY_DIR}/VTK
 ${PYTHON_INCLUDE_PATH}
)

ADD_LIBRARY(pqPython
  pqPythonExport.h
  pqPythonInterpreter.cxx
  pqPythonInterpreter.h
)

IF (WIN32)
  TARGET_LINK_LIBRARIES(pqPython
    ${PYTHON_LIBRARIES}
  )
ELSE (WIN32)
  TARGET_LINK_LIBRARIES(pqPython
    ${PYTHON_LIBRARIES}
    ${PYTHON_UTIL_LIBRARY}
  )
ENDIF (WIN32)
IF(NOT PV_INSTALL_NO_LIBRARIES)
  SET (targets_to_install pqPython)
    
  IF(PV_INSTALL_HAS_CMAKE_24)
    INSTALL(TARGETS ${targets_to_install}
      RUNTIME DESTINATION ${PV_INSTALL_BIN_DIR_CM24} COMPONENT Runtime
      LIBRARY DESTINATION ${PV_INSTALL_LIB_DIR_CM24} COMPONENT Runtime
      ARCHIVE DESTINATION ${PV_INSTALL_LIB_DIR_CM24} COMPONENT Development)
  ELSE(PV_INSTALL_HAS_CMAKE_24)
    INSTALL_TARGETS(${PV_INSTALL_LIB_DIR} ${targets_to_install})
  ENDIF(PV_INSTALL_HAS_CMAKE_24)

ENDIF(NOT PV_INSTALL_NO_LIBRARIES)
